// --- 修正版：摘みの制御ロジック（真っ白画面回避） --- const startAction = (clientX) => { startX = clientX; startTime = Date.now(); if (sidebar.classList.contains('hidden')) return; isResizing = true; toggleBtn.classList.add('active'); sidebar.style.transition = 'none'; toggleBtn.style.transition = 'none'; document.body.style.cursor = 'col-resize'; }; const doAction = (clientX) => { if (!isResizing) return; let newWidth = clientX; // 150pxの壁：物理的に止める if (newWidth < 150) newWidth = 150; if (newWidth > window.innerWidth * 0.8) newWidth = window.innerWidth * 0.8; currentWidth = newWidth; sidebar.style.width = currentWidth + 'px'; toggleBtn.style.left = currentWidth + 'px'; // 時計の可変サイズ（3.5rem上限） const fontSize = Math.min(3.5, currentWidth / 80); digitalClock.style.fontSize = fontSize + 'rem'; }; const endAction = (clientX) => { // clientX が undefined の場合のフォールバック const endX = (clientX !== undefined) ? clientX : startX; const duration = Date.now() - startTime; const moveX = Math.abs(endX - startX); // タップ判定：10px以内の移動かつ300ms以内なら開閉 if (moveX < 10 && duration < 300) { if (sidebar.classList.contains('hidden')) { openSidebar(); } else { closeSidebar(); } } isResizing = false; toggleBtn.classList.remove('active'); document.body.style.cursor = 'default'; if (!sidebar.classList.contains('hidden')) { sidebar.style.transition = 'transform 0.3s ease, margin-left 0.3s ease'; toggleBtn.style.transition = 'left 0.3s ease, background 0.2s'; } keepInBounds(); }; // Mouse Events toggleBtn.addEventListener('mousedown', (e) => startAction(e.clientX)); window.addEventListener('mousemove', (e) => doAction(e.clientX)); window.addEventListener('mouseup', (e) => endAction(e.clientX)); // Touch Events (真っ白画面対策) toggleBtn.addEventListener('touchstart', (e) => { if (e.touches && e.touches[0]) { startAction(e.touches[0].clientX); } }, {passive: false}); window.addEventListener('touchmove', (e) => { if (isResizing && e.touches && e.touches[0]) { doAction(e.touches[0].clientX); e.preventDefault(); } }, {passive: false}); window.addEventListener('touchend', (e) => { // changedTouches がない場合のエラーを回避 const touchX = (e.changedTouches && e.changedTouches[0]) ? e.changedTouches[0].clientX : undefined; endAction(touchX); });
