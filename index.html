// --- PCマウス用：クリック中のみ動かす制御 --- const handleMouseMove = (e) => { if (isResizing) onMove(e.clientX); }; const handleMouseUp = () => { window.removeEventListener('mousemove', handleMouseMove); window.removeEventListener('mouseup', handleMouseUp); onEnd(); }; toggleBtn.addEventListener('mousedown', (e) => { onStart(e.clientX); // クリックされた時だけ、ウィンドウ全体の動きを監視する window.addEventListener('mousemove', handleMouseMove); window.addEventListener('mouseup', handleMouseUp); }); // --- タッチ用（既存） --- toggleBtn.addEventListener('touchstart', (e) => onStart(e.touches[0].clientX), {passive: false}); window.addEventListener('touchmove', (e) => { if(isResizing) { onMove(e.touches[0].clientX); e.preventDefault(); } }, {passive: false}); window.addEventListener('touchend', onEnd);
